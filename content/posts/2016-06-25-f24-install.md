Repos:  
[bash]  
cat << EOF > /etc/yum.repos.d/google-chrome.repo  
[google-chrome]  
name=google-chrome &#8211; \$basearch  
baseurl=http://dl.google.com/linux/chrome/rpm/stable/\$basearch  
enabled=1  
gpgcheck=1  
gpgkey=https://dl-ssl.google.com/linux/linux\_signing\_key.pub  
EOF

[/bash]
